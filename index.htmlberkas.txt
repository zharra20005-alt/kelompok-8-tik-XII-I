  <div class="container">
    <h1>Tool: Embed / Open Gemini Share</h1>
    <p>Masukkan link Gemini Share di bawah lalu pilih tindakan. Contoh link sudah terisi otomatis.</p>

    <label for="url">Gemini Share URL</label>
    <input id="url" type="text" value="https://gemini.google.com/share/aff5a700f9a7">

    <div class="row">
      <button id="openBtn">Buka di Tab Baru</button>
      <button id="embedBtn">Tampilkan di Halaman (iframe)</button>
      <button id="copyBtn" class="secondary">Salin Link</button>
      <button id="downloadBtn" class="secondary">Coba Unduh (fetch)</button>
    </div>

    <div id="frameWrap"></div>

    <div class="note">
      <strong>Catatan:</strong>
      <ul>
        <li>Beberapa situs (termasuk layanan modern) mungkin menolak ditampilkan dalam &lt;iframe&gt; karena kebijakan keamanan (X-Frame-Options / Content-Security-Policy).</li>
        <li>Tombol unduh menggunakan <code>fetch</code> dan bisa gagal karena CORS. Jika gagal, buka langsung di tab baru lalu simpan manual.</li>
      </ul>
    </div>
  </div>

  <script>
    const urlInput = document.getElementById('url');
    const openBtn = document.getElementById('openBtn');
    const embedBtn = document.getElementById('embedBtn');
    const copyBtn = document.getElementById('copyBtn');
    const downloadBtn = document.getElementById('downloadBtn');
    const frameWrap = document.getElementById('frameWrap');

    openBtn.addEventListener('click', ()=>{
      const u = urlInput.value.trim();
      if(!u) return alert('Masukkan URL terlebih dahulu');
      window.open(u, '_blank');
    });

    embedBtn.addEventListener('click', ()=>{
      const u = urlInput.value.trim();
      if(!u) return alert('Masukkan URL terlebih dahulu');
      frameWrap.innerHTML = '';
      const iframe = document.createElement('iframe');
      iframe.src = u;
      iframe.loading = 'lazy';
      frameWrap.appendChild(iframe);
    });

    copyBtn.addEventListener('click', async ()=>{
      const u = urlInput.value.trim();
      if(!u) return alert('Masukkan URL terlebih dahulu');
      try{
        await navigator.clipboard.writeText(u);
        alert('Link disalin ke clipboard');
      }catch(e){
        alert('Gagal menyalin. Silakan salin manual.');
      }
    });

    downloadBtn.addEventListener('click', async ()=>{
      const u = urlInput.value.trim();
      if(!u) return alert('Masukkan URL terlebih dahulu');
      try{
        downloadBtn.disabled = true;
        downloadBtn.textContent = 'Mengunduh...';
        const res = await fetch(u);
        if(!res.ok) throw new Error('Fetch gagal: ' + res.status);
        const blob = await res.blob();
        const a = document.createElement('a');
        const url = URL.createObjectURL(blob);
        a.href = url;
        a.download = 'gemini-share';
        document.body.appendChild(a);
        a.click();
        a.remove();
        URL.revokeObjectURL(url);
        alert('Jika tidak ada file yang diunduh, server mungkin menolak permintaan (CORS) atau tidak menyediakan file langsung.');
      }catch(err){
        alert('Gagal mengunduh: ' + err.message + '\nSolusi: klik "Buka di Tab Baru" lalu simpan manual.');
      }finally{
        downloadBtn.disabled = false;
        downloadBtn.textContent = 'Coba Unduh (fetch)';
      }
    });
  </script>
